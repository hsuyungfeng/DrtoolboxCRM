```typescript
@Injectable()
export class FeeCalculationService {
  constructor(
    @InjectRepository(Payment) private paymentRepo: Repository<Payment>,
    @InjectRepository(Treatment) private treatmentRepo: Repository<Treatment>,
  ) {}

  /**
   * 計算療程費用餘額（未付金額）
   * @returns balance = finalPrice - sum(payments)，最小為 0
   */
  async calculateBalance(treatmentId: string, clinicId: string): Promise<{
    totalFee: number;
    totalPaid: number;
    balance: number;
    payments: Payment[];
  }> {
    // 使用 finalPrice（已扣積點），Decimal.js 精確計算
    const treatment = await this.treatmentRepo.findOne({
      where: { id: treatmentId, clinicId }
    });
    if (!treatment) throw new NotFoundException(...);

    const payments = await this.paymentRepo.find({
      where: { treatmentId, clinicId, status: 'completed' },
      order: { paidAt: 'ASC' }
    });

    const totalFee = new Decimal(treatment.finalPrice ?? treatment.totalPrice);
    const totalPaid = payments.reduce(
      (sum, p) => sum.plus(new Decimal(p.amount)),
      new Decimal(0)
    );
    const balance = Decimal.max(totalFee.minus(totalPaid), 0);

    return {
      totalFee: totalFee.toDecimalPlaces(2).toNumber(),
      totalPaid: totalPaid.toDecimalPlaces(2).toNumber(),
      balance: balance.toDecimalPlaces(2).toNumber(),
      payments,
    };
  }

  /**
   * 計算療程總費用（totalPrice × sessionCount 已在 Treatment 記錄）
   * FIN-01：返回 finalPrice（扣積點後）與費用明細
   */
  async calculateTreatmentFee(treatmentId: string, clinicId: string): Promise<{
    totalPrice: number;
    pointsRedeemed: number;
    finalPrice: number;
  }> {
    const treatment = await this.treatmentRepo.findOne({
      where: { id: treatmentId, clinicId }
    });
    if (!treatment) throw new NotFoundException(...);
    return {
      totalPrice: new Decimal(treatment.totalPrice).toDecimalPlaces(2).toNumber(),
      pointsRedeemed: new Decimal(treatment.pointsRedeemed ?? 0).toDecimalPlaces(2).toNumber(),
      finalPrice: new Decimal(treatment.finalPrice ?? treatment.totalPrice).toDecimalPlaces(2).toNumber(),
    };
  }
}
```

建立 `backend/src/revenue/services/payment.service.ts`：

```typescript
@Injectable()
export class PaymentService {
  constructor(
    @InjectRepository(Payment) private paymentRepo: Repository<Payment>,
    private feeCalculationService: FeeCalculationService,
  ) {}

  async create(dto: CreatePaymentDto, clinicId: string): Promise<Payment> {
    const payment = this.paymentRepo.create({
      ...dto,
      clinicId,
      paidAt: dto.paidAt ? new Date(dto.paidAt) : new Date(),
      status: 'completed',
    });
    return await this.paymentRepo.save(payment);
  }

  async findByTreatment(treatmentId: string, clinicId: string): Promise<Payment[]> {
    return this.paymentRepo.find({
      where: { treatmentId, clinicId },
      order: { paidAt: 'ASC' },
    });
  }

  async findByPatient(patientId: string, clinicId: string): Promise<Payment[]> {
    return this.paymentRepo.find({
      where: { patientId, clinicId },
      order: { paidAt: 'DESC' },
      relations: ['treatment'],
    });
  }

  async getBalance(treatmentId: string, clinicId: string) {
    return this.feeCalculationService.calculateBalance(treatmentId, clinicId);
  }

  async remove(id: string, clinicId: string): Promise<void> {
    const payment = await this.paymentRepo.findOne({ where: { id, clinicId } });
    if (!payment) throw new NotFoundException(`Payment ${id} not found`);
    if (payment.status === 'completed') {
      // 軟刪除：標記 cancelled
      payment.status = 'cancelled';
      await this.paymentRepo.save(payment);
    }
  }
}
```
